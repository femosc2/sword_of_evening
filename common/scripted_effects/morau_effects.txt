agot_create_artifact_vs_twilight_effect = {
	$OWNER$ = {
		save_scope_as = owner
	}
	set_artifact_rarity_illustrious = yes
	create_artifact = {
		name = vs_twilight_name
		visuals = vs_dawn_visuals
		description = vs_twilight_description
		type = sword
		wealth = scope:wealth
		quality = scope:quality
		template = vs_twilight_template
		modifier = vs_twilight_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact ?= {
		if = {
			limit = {
				scope:owner = { has_character_flag = is_sword_of_evening }
			}
			equip_artifact_to_owner_replace = yes
		}
		set_variable = {
			name = valyrian_steel
			value = yes
		}
	}
}
